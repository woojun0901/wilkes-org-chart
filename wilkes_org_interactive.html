  <script>

    function updatePath(node) {

      const pathEl = document.getElementById('path');

      const labels = [];

      let current = node;

      while (current && current.classList && current.classList.contains('node')) {

        const lbl = current.getAttribute('data-label');

        if (lbl) labels.unshift(lbl);

        current = current.parentElement.closest('.node');

      }

      if (!labels.length) {

        pathEl.innerHTML = 'Current selection: <strong>Wilkes University</strong>';

      } else {

        pathEl.innerHTML = 'Current selection: <strong>' + labels.join(' \\ ') + '</strong>';

      }

    }


    function toggleNode(node) {

      const children = node.querySelector(':scope > .children');

      if (!children) return;

      children.classList.toggle('hidden');

      updatePath(node);

    }


    document.addEventListener('click', function(e) {

      const label = e.target.closest('.label');

      if (!label) return;

      const node = label.parentElement;

      if (!node.classList.contains('node')) return;

      toggleNode(node);

    });


    function setAll(expand) {

      document.querySelectorAll('.children').forEach(ch => {

        if (expand) {

          ch.classList.remove('hidden');

        } else {

          ch.classList.add('hidden');

        }

      });

      const uni = document.querySelector('.level-university .children');

      if (uni) uni.classList.remove('hidden');

      document.querySelectorAll('.level-college .children').forEach(ch => ch.classList.remove('hidden'));

      const root = document.querySelector('.level-university');

      if (root) updatePath(root);

    }


    document.getElementById('expandAll').addEventListener('click', function() { setAll(true); });

    document.getElementById('collapseAll').addEventListener('click', function() { setAll(false); });


    setAll(false);

  </script>
